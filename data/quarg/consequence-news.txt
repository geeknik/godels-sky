# QUARG TRADER PATTERN REACTIONS
# =============================================================================

	to show
		"actionlog: pattern" == 1
		"reputation: Quarg" > 10
	location
		government "Quarg"
	name
		word
			"Quarg observer"
			"Quarg"
	message
		word
			`"`
		word
			"You are what is called a 'merchant.' Your species develops through trade. This is known to us."
			"We have observed your trading patterns. They are consistent with peaceful development. This is acceptable."
			"Trade is how young species learn cooperation. You practice this well. It is a favorable sign for your kind."
			"Your cargo manifests show fair exchange. This is the foundation of civilization. You build well."
			"Many of your species resort to violence for resources. You choose trade. This is the wiser path."
		word
			`"`
